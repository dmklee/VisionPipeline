ALG 1 - ExtractContours
        INPUTS:  single-channel image
        OUTPUTS: vector of contour locations

    image = suppressNoise(image)
    edgeMap, gradientMap = computeEdgeAndGradMap(image)
    seeds = extractSeeds(edgeMap, gradientMap)
    contour_locations <- []
    for seed in seeds:
       shiftSeed(seed, edgeMap, gradientMap)
       if !isStableSeed(seed, edgeMap, gradientMap):
            continue
       contour, model, success = growSeed(seed, edgeMap, gradientMap)
        if !success:
            continue
        location, contour_type = localizeContour(contour, contour_type)
        contour_locations.append(location)

    return contour_locations
